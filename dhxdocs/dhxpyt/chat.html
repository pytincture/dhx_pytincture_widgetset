260    def _extract_content_text(content: Any) -> str:
261        if isinstance(content, str):
262            return content
263        if isinstance(content, list):
264            parts = []
265            for item in content:
266                if isinstance(item, str):
267                    parts.append(item)
268                elif isinstance(item, dict):
269                    text = item.get("text")
270                    if isinstance(text, str):
271                        parts.append(text)
272                        continue
273                    text = item.get("content")
274                    if isinstance(text, str):
275                        parts.append(text)
276            return "".join(parts)

279    @staticmethod
280    def extract_stream_text(chunk: Any) -> str:
281        if chunk is None:
282            return ""
283
284        if isinstance(chunk, str):
285            return chunk
286
287        payload = chunk
288        if hasattr(payload, "model_dump"):
289            payload = payload.model_dump(exclude_none=True)
290        elif hasattr(payload, "to_dict"):
291            payload = payload.to_dict()
292
293        if isinstance(payload, str):
294            return payload
295
296        if not isinstance(payload, dict):
297            return ""
298
299        chunk_type = payload.get("type")
300
301        if chunk_type == "chunk":
302            payload = payload.get("chunk", {})
303            choices = payload.get("choices") or []
304        elif chunk_type == "content.delta":
305            return ""
306        elif chunk_type == "response.output_text.delta":
307            delta = payload.get("delta")
308            if isinstance(delta, dict):
309                return delta.get("content") or delta.get("text") or ""
310            return delta or ""
311        else:
312            choices = payload.get("choices")
313            if choices is None and "chunk" in payload:
314                choices = payload["chunk"].get("choices")
315
316        if not choices:
317            if "content" in payload:
318                return Chat._extract_content_text(payload.get("content"))
319            return ""
320
321        for choice in choices:
322            delta = (choice or {}).get("delta")
323            if not delta:
324                continue
325            if isinstance(delta, str):
326                return delta
327            if isinstance(delta, dict):
328                text = delta.get("content")
329                if text:
330                    return Chat._extract_content_text(text)
331                text = delta.get("text")
332                if isinstance(text, str):
333                    return text
334        return ""

336    def consume_stream(
337        self,
338        response_id: str,
339        stream: Any,
340        *,
341        parser: Optional[Callable[[Any], str]] = None,
342        finish: bool = True,
343        on_error: Optional[Callable[[Exception], None]] = None,
344    ) -> None:
345        tokenize = parser or self.extract_stream_text
346
347        def handle_error(exc: Exception) -> None:
348            if on_error:
349                on_error(exc)
350                return
351            warning = f"Backend error: {exc}"
352            try:
353                self.append_stream(response_id, warning)
354            finally:
355                self.finish_stream(response_id)
356            logging.warning("[Chat] stream failed: %s", exc)
357
358        if inspect.isasyncgen(stream) or (
359            hasattr(stream, "__aiter__") and not hasattr(stream, "__iter__")
360        ):
361
362            async def runner():
363                try:
364                    async for chunk in stream:
365                        text = tokenize(chunk)
366                        if text:
367                            self.append_stream(response_id, text)
368                except Exception as exc:  # pragma: no cover - pass to handler
369                    handle_error(exc)
370                else:
371                    if finish:
372                        self.finish_stream(response_id)
373
374            self._run_async(runner())
375            return
376
377        try:
378            iterator = iter(stream)
379        except TypeError:
380            raise ValueError("stream must be an iterable or async iterable") from None
381
382        try:
383            for chunk in iterator:
384                text = tokenize(chunk)
385                if not text:
386                    continue
387                self.append_stream(response_id, text)
388        except Exception as exc:  # pragma: no cover - pass to handler
389            handle_error(exc)
390            return
391
392        if finish:
393            self.finish_stream(response_id)

395    @staticmethod
396    def _run_async(coro):
397        try:
398            loop = asyncio.get_running_loop()
399        except RuntimeError:
400            loop = None
401
402        if loop and loop.is_running():
403            loop.create_task(coro)
404        else:
405            asyncio.run(coro)
